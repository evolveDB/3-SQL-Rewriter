		(select osi.order_id,osi.sub_order_id,osi.sub_order_status,oc.order_finish_time_limit,oi.time_limit_return,oi.time_limit_replacement,osi.confirm_time
		from order_sub_info osi INNER JOIN order_config oc on osi.shop_id=oc.shop_id left join order_item oi on osi.sub_order_id=oi.sub_order_id
		where oc.config_type='02' and osi.order_channel='40' and osi.sub_order_status='05' and oi.item_status='0' and osi.sub_order_cat!='68'
		UNION
		select osi.order_id,osi.sub_order_id,osi.sub_order_status,oc.order_finish_time_limit,oi.time_limit_return,oi.time_limit_replacement,osi.confirm_time
		from order_sub_info osi INNER JOIN order_config oc on osi.province_info=oc.province_info left join order_item oi on osi.sub_order_id=oi.sub_order_id
		where oc.config_type='01' and osi.order_channel!='40' and osi.sub_order_status='05' and osi.sub_order_cat!='68'
		) as t
		where t.confirm_time<=DATE_SUB(NOW(),INTERVAL t.order_finish_time_limit DAY)
		group by t.order_id,t.sub_order_id,t.order_finish_time_limit,t.confirm_time
		having
		confirm_time<=DATE_SUB(NOW(),INTERVAL time_limit_return DAY)
		and confirm_time<=DATE_SUB(NOW(),INTERVAL time_limit_replacement DAY);

SELECT
        oin.order_id AS orderNo,
        oin.order_channel AS orderChannel,
        oin.order_type AS orderType,
        '售' AS businessType,
        osin.sub_order_status AS orderStatus,
        osin.third_party_order_id AS jdOrderId,
        oin.create_time AS startTime,
        oin.pay_time AS payTime,
        oin.serial_id AS transactionSerialNumber,
        (CASE
        WHEN oin.consumer_payment_type = '0' THEN '现金'
        WHEN oin.consumer_payment_type = '1' THEN '微信'
        WHEN oin.consumer_payment_type = '2' THEN '支付宝'
        WHEN oin.consumer_payment_type = '4' THEN '-'
        WHEN oin.consumer_payment_type = '3' THEN '其他'
        END) AS paymentMethod,
        oit.discount_name AS discountWay,
        oit.deduction_name AS deductionWay,
        CONVERT( (IFNULL(oit.discount_amount,0)/oit.number) + IFNULL(oit.deduction_amount_by_coupons, 0) + IFNULL(oit.split_coupon_amount, 0),DECIMAL(15,2)) AS discountMoney,
        CONVERT( oit.deduction/oit.number  ,DECIMAL(15,2)) AS deductionMoney,

        convert((oit.original_total_price-oit.discount_amount-oit.deduction)/oit.number,DECIMAL(15,2)) AS factTotalPrice,

        oit.brand_ch_name AS brand,
        oit.class_all_level_name AS goodsClass,
        oit.spu_name AS goodsName,
        oit.sku_name AS goodsSpecification,
        oit.tax_rate AS taxRate,
        oit.sku_id AS specificationNo,
        osse.supplier_ch_name AS supplier,
        oit.number AS noInventoryNum,
        IFNULL(osse.number,oit.number) AS num,
        osse.serial_no AS snNo,
        oit.manager_style AS inventoryManageType,
        oit.original_price AS saleUnitPrice,
        case when osse.cost_price is null or osse.cost_price = '' then oit.cost_price else osse.cost_price end AS costPrice,
        oin.order_note AS orderRemark,
        osin.handle_note AS subOrderRemark,
        osin.sub_customer_note AS customerRemark,
        osin.employee_dept_code AS saleEmployeeDeptCode,
        osin.employee_dept_name AS orderDeptName,
        CONCAT_WS('/',csmd.parent_name,csmd.name) AS saleEmployeeDeptName,
        osin.employee_name AS saleEmployee,
        osin.payee_name AS operator,
        osin.employee_no AS saleEmployeeNo,
        osin.shop_name AS shopName,
        osin.store_id AS storeId,
        oain.broadband_account AS broadbandAccount,
        IFNULL(osin.sub_customer_note,'') AS remark,
        oain.customer_type AS saleWay,
        IF(oit.manager_style = '1' AND oain.customer_type = '02',oain.enterprise_name,oain.consignee_realname) AS customerName,
        IFNULL(oain.consignee_telphone,'') AS customerPhone,
        CONCAT(
        IFNULL(oain.customer_country, ''),
        IFNULL(oain.customer_privince, ''),
        IFNULL(oain.customer_city, ''),
        IFNULL(oain.customer_area, ''),
        IFNULL(oain.customer_street, ''),
        IFNULL(oain.consignee_adress, '')
        ) AS customerAddress,
        oin.is_invoices AS invoice,
        fsse.subject_full_name AS classSubjectName,
        fsse.subject_full_code AS classSubjectCode,
        fsse1.subject_full_name AS stockSubjectName,
        fsse1.subject_full_code  AS stockSubjectCode,
        oit.remark2 AS tag1,
        oit.remark3 AS tag2,
        oit.remark4 AS tag3,
        osse.batch_number_new as batchNumberNew,
        osin.sub_order_item_count as subOrderItemCount,
        oit.number as orderItemBuyNumber,
        osin.sub_order_logistics_fee as subOrderLogisticsFee
        FROM order_item oit
        LEFT JOIN order_info oin ON oit.order_id = oin.order_id
        LEFT JOIN order_sub_info osin ON oit.sub_order_id = osin.sub_order_id
        LEFT JOIN order_sale_serial osse ON osse.order_item_id = oit.order_item_id
        LEFT JOIN order_auditbiz_info oain ON oain.sub_order_id = osin.sub_order_id
        LEFT JOIN authentication.csm_employee cem ON cem.user_id = osin.employee_id
        LEFT JOIN authentication.csm_dept csmd ON csmd.codenew = cem.dept_code
        LEFT JOIN statistics.financial_subject_setting fsse ON fsse.relation_detail = oit.class_all_level_name AND fsse.relation_info = 2
        LEFT JOIN statistics.financial_subject_setting fsse1 ON fsse1.relation_detail_id = osse.stock_type AND fsse1.relation_info = 1
         
        WHERE oin.pay_time >= '2022-09-21 00:00:00' AND oin.pay_time <= '2022-11-25 23:59:59'
        AND osin.sub_order_type IN ('01','02','05')
         
             
                    AND osin.employee_dept_code LIKE "_"'771'"%"
                 
         
         
         
         
         
         
         
         
         
         
         
         
         
         
         
         
         
         
         
         
         
     
        ORDER BY oin.pay_time desc
         
            LIMIT 0,50;

        case when osse.cost_price is null or osse.cost_price = '' then oit.cost_price else osse.cost_price end AS costPrice,
        oin.order_note AS orderRemark,
        osin.handle_note AS subOrderRemark,
        osin.sub_customer_note AS customerRemark,
        osin.employee_dept_code AS saleEmployeeDeptCode,
        osin.employee_dept_name AS orderDeptName,
        CONCAT_WS('/',csmd.parent_name,csmd.name) AS saleEmployeeDeptName,
        osin.employee_name AS saleEmployee,
        osin.payee_name AS operator,
        osin.employee_no AS saleEmployeeNo,
        osin.shop_name AS shopName,
        osin.store_id AS storeId,
        oain.broadband_account AS broadbandAccount,
        IFNULL(osin.sub_customer_note,'') AS remark,
        oain.customer_type AS saleWay,
        IF(oit.manager_style = '1' AND oain.customer_type = '02',oain.enterprise_name,oain.consignee_realname) AS customerName,

        IFNULL(oain.customer_country, ''),
        IFNULL(oain.customer_privince, ''),
        IFNULL(oain.customer_city, ''),
        IFNULL(oain.customer_area, ''),
        IFNULL(oain.customer_street, ''),

        oit.remark4 AS tag3,

                    AND osin.employee_dept_code LIKE '3571579'"%"
                 
         
         
         
         
         
         
            AND oit.spu_name LIKE "%"'投影仪'"%"

            LIMIT 0,50;
